#include < iostream >
using namespace std;

int main() {
    int n, quantity;
    string item;
    double price, total = 0, amount;

    cout << "Enter number of items: ";
    cin >> n;

    cin.ignore();

    for (int i = 1; i <= n; i++) {
        cout << "\nItem " << i << endl;

        cout << "Name: ";
        getline(cin, item);

        cout << "Price: ";
        cin >> price;

        cout << "Quantity: ";
        cin >> quantity;

        amount = price * quantity;
        total += amount;

        cin.ignore();
    }

    cout << "\nTotal Payment = $" << total << endl;

    return 0;
}
