មេរៀនទី២៖ Variables ក្នុង C++

១. តើ Variable ជាអ្វី?

Variable គឺជា កន្លែងផ្ទុកទិន្នន័យ (Data) នៅក្នុងកម្មវិធី។ យើងប្រើ Variable ដើម្បីរក្សាទុកតម្លៃផ្សេងៗ ដូចជា លេខ អក្សរ ឬអត្ថបទ។

២. របៀបប្រកាស Variable (Syntax)

    dataType variableName = value;

ឧទាហរណ៍៖

    int age = 18;
  • int = ប្រភេទទិន្នន័យ
  • age = ឈ្មោះ Variable
  • 18 = តម្លៃដែលផ្ទុក

៣. ប្រភេទ Variable ដែលប្រើញឹកញាប់

ប្រភេទ ការប្រើប្រាស់ ឧទាហរណ៍
int ផ្ទុកលេខគត់ 10, 25, -5
bool ផ្ទុកតម្លៃពិត/មិនពិត true, false
string ផ្ទុកអត្ថបទ "Hong"
char ផ្ទុកតួអក្សរតែមួយ 'A', 'B'
double ផ្ទុកលេខទសភាគ 3.14, 19.99

៤. ឧទាហរណ៍កូដ

#include < iostream >
using namespace std;

int main() {
    int age = 18;
    double score = 95.5;
    char grade = 'A';
    string name = "Hong";
    bool pass = true;

    cout << "Name: " << name << endl;
    cout << "Age: " << age << endl;
    cout << "Score: " << score << endl;
    cout << "Grade: " << grade << endl;
    cout << "Pass: " << pass << endl;

    return 0;
}

លទ្ធផល (Output)

    Name: Hong
Age: 18
Score: 95.5
Grade: A
Pass: 1

៥. ច្បាប់ក្នុងការដាក់ឈ្មោះ Variable

✅ អាចប្រើអក្សរ លេខ និង _

studentName
student_age
score1

❌ មិនអាចចាប់ផ្តើមដោយលេខ

    1score    // ខុស

❌ មិនអាចប្រើ Keyword របស់ C++

   int int = 10;   // ខុស

៦. ការផ្លាស់ប្តូរតម្លៃ Variable

#include < iostream >
using namespace std;

int main() {
    int number = 10;

    cout << number << endl;

    number = 20;

    cout << number << endl;

    return 0;
}

លទ្ធផល

10
20
